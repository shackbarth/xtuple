  2 /** @namespace
  3   Exception handling for the process.
  4   
  5   @todo Complete the implementation
  6 */
  7 xt.exception = xt.object.create(
  8   /** @lends xt.exception.prototype */ {
  9     
 10   /**
 11     All manageable exceptions encountered during runtime
 12     will be handed to this method to be dealt with properly.
 13     An appropriately thrown error will minimally have
 14     a type and a message and can have an array of arguments to
 15     be applied to the handler for that type.
 16     
 17     @param {Object} e The object hash returned from the exception.
 18     @private
 19   */
 20   handle: function(e) {
 21     var s = xt.exception;
 22     switch(e.type) {
 23       case xt.exception.t_warning:
 24       case xt.exception.t_report:
 25         xt.warn("Exception encountered", e);
 26         break;
 27       case xt.exception.t_close:
 28         var a = e.args;
 29         if(xt.kindOf(a[0], xt.response))
 30           a[0].close();
 31         delete e.args;
 32         xt.warn("Exception required closing client connection", e);
 33         break;
 34       case xt.exception.t_fatal:
 35         xt.err("Error encountered", e);
 36         s.count += 1;
 37         break;
 38       default:
 39         xt.err("Unrecoverable exception", e);
 40         xt.trucking = NO;
 41         break;
 42     }
 43     if(s.count > 5) throw xt.fatal("Too many errors encountered");
 44   },
 45     
 46   //...........................................
 47   // Types of Exceptions
 48   //
 49 
 50   /** @property */
 51   t_fatal:    'fatal',
 52 
 53   /** @property */
 54   t_warning:  'warning',
 55 
 56   /** @property */
 57   t_close:    'close',
 58 
 59   /** @property */
 60   t_report:   'report',
 61   
 62   /**
 63     Creates a fatal error hash from the message.
 64 
 65     @param {String} message The message for the exception.
 66     @return {Object} The exception hash.
 67   */
 68   fatal: function() {
 69     var m = arguments[0];
 70     return { type: 'fatal', message: m };
 71   },
 72   
 73   /**
 74     Creates a warning hash from the message.
 75 
 76     @param {String} message The message for the exception.
 77     @return {Object} The exception hash.
 78   */
 79   warning: function() {
 80     var m = arguments[0];
 81     return { type: 'warning', message: m };
 82   },
 83   
 84   /**
 85     Creates a close hash from the message and any additional
 86     arguments.
 87 
 88     @param {String} message The message for the exception.
 89     @param {Object} [args] Any arguments to be passed to the handler.
 90     @return {Object} The exception Hash.
 91   */
 92   close: function() {
 93     var a = args(),
 94         m = a.shift();
 95     return { type: 'close', message: m, args: a };
 96   },
 97   
 98   count: 0,
 99   
100   /** @private */
101   className: 'xt.exception'
102     
103 }) ;

108 xt.close    = xt.exception.close;

119 /**
120   Much like native throw, issue will pass an xt exception directly to the
121   handler. If the exception is fatal it will cease execution of the entire
122   system. If it is non-fatal, it will handle the exception and continue
123   executing in the current context where it was issued.
124 
125   Note that issue is the prefered way to handle exceptions that interrupt
126   responses to HTTP requests in a server.
127 
128   @see xt#exception#close
129 
130   @param {Object} exception The xt exception to issue.
131   @param {Object} [response] Any active server response to be affected.
132 */
133 issue = function() {
134   var a = args();
135   if(!a || a.length <= 0) return;
136   xt.exception.handle(a[0]);
137 }
